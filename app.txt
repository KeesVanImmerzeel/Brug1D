#' Plot de horizontale fluxverandering q [L2/T] voor verschillende tijdstippen t
#'
#' @param df Dataframe met de resultaten van calc_stress_response
#' @example plot_stress_response_q(df = result_calc_stress_response_n0)
#' @return Een ggplot object
plot_stress_response_q <- function(df) {
  title_text <- switch(as.character(df$n[1]),
                       "0" = paste("Horizontale fluxverandering q [L2/T]: Rivierstand verandert plotseling met een vaste waarde a=", df$a[1], "[L]"),
                       "1" = paste("Horizontale fluxverandering q [L2/T]: Constante infiltratie a=", df$a[1], "[L2/T] start op x = 0"),
                       "2" = paste("Horizontale fluxverandering q [L2/T]: Rivierstand stijgt met constante snelheid a=", df$a[1], "[L/T]"),
                       "3" = paste("Horizontale fluxverandering q [L2/T]: Infiltratie op x = 0 neemt met constante snelheid a=", df$a[1], "[L2/T] toe"),
                       paste("Onbekende stress verandering"))
  
  ggplot(df, aes(x = x, y = q, color = factor(t))) +
    geom_line(linewidth = 1.5) +
    labs(title = title_text,
         x = "Afstand vanaf de rivier [L]",
         y = "Horizontale fluxverandering q [L2/T]",
         color = "Tijd [T]") +
    theme_minimal() +
    theme(axis.text = element_text(size = 14),
          axis.title = element_text(size = 16, margin = margin(t = 30)),
          legend.text = element_text(size = 14),
          legend.title = element_text(size = 16, margin = margin(b = 20)),
          panel.border = element_rect(color = "black", fill = NA, size = 0.5),
          legend.box.background = element_rect(color = "black", size = 0.5))
}
